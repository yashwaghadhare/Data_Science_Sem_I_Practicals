# Step 1: Install Pandas (if not installed already)
# Run this in terminal or command prompt:
# !pip install pandas
# Step 2: Import Pandas
import pandas as pd
# Step 3: Create Sample Data
data = {
    'Name': ['Alice', 'Bob', 'Charlie'],
    'Age': [25, 30, 28],
    'City': ['Mumbai', 'Delhi', 'Bangalore']
}
df = pd.DataFrame(data)
print("Original DataFrame:")
print(df)
# Step 4: Save Data to CSV File
df.to_csv('data.csv', index=False)  # index=False avoids saving row numbers
# Step 5: Read Data Back from CSV
df2 = pd.read_csv('data.csv')
print("\nDataFrame after reading from CSV:")
print(df2)
